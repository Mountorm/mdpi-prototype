# Cloudflare Workers 网页部署指南

本指南将帮助你通过 Cloudflare 网页控制台手动部署 Mock API 服务。

## 📋 前提条件

- 一个 Cloudflare 账号（免费账号即可）
- 如果需要自定义域名，需要在 Cloudflare 托管你的域名

---

## 🚀 部署步骤

### 步骤 1: 登录 Cloudflare

1. 访问 [https://dash.cloudflare.com/](https://dash.cloudflare.com/)
2. 使用你的账号登录

### 步骤 2: 进入 Workers 页面

1. 在左侧菜单中，点击 **Workers & Pages**
2. 点击 **Create application** 按钮
3. 选择 **Create Worker** 标签

### 步骤 3: 创建 Worker

1. 给你的 Worker 起一个名字，例如：`mock-api-service`
   - 这个名字将成为你的默认域名：`mock-api-service.workers.dev`
2. 点击 **Deploy** 按钮（先创建，稍后修改代码）

### 步骤 4: 编辑 Worker 代码

1. 部署完成后，点击 **Edit code** 按钮
2. 删除默认的示例代码
3. 复制下面的完整代码，粘贴到编辑器中

### 步骤 5: 复制代码

**完整的 Worker 代码：**

```javascript
/**
 * Mock API Service for Cloudflare Workers
 * 模拟三个 API 接口
 */

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);
    const path = url.pathname;
    const method = request.method;

    // CORS 头设置
    const corsHeaders = {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
      'Access-Control-Allow-Headers': 'Content-Type',
    };

    // 处理 OPTIONS 预检请求
    if (method === 'OPTIONS') {
      return new Response(null, {
        headers: corsHeaders,
      });
    }

    try {
      // 路由处理
      if (method === 'POST' && path === '/api/v2/ingest/submission') {
        return handleIngestSubmission(request, corsHeaders);
      }

      if (method === 'GET' && path.match(/^\/api\/v2\/documents\/[^/]+\/status$/)) {
        const documentHash = path.split('/')[4];
        return handleDocumentStatus(documentHash, corsHeaders);
      }

      if (method === 'GET' && path.match(/^\/api\/v2\/reports\/[^/]+\/issues_summary$/)) {
        const documentHash = path.split('/')[4];
        return handleIssuesSummary(documentHash, corsHeaders);
      }

      // 404 处理
      return new Response(
        JSON.stringify({
          error: 'Not Found',
          message: `Path ${path} not found`,
        }),
        {
          status: 404,
          headers: {
            'Content-Type': 'application/json',
            ...corsHeaders,
          },
        }
      );
    } catch (error) {
      return new Response(
        JSON.stringify({
          error: 'Internal Server Error',
          message: error.message,
        }),
        {
          status: 500,
          headers: {
            'Content-Type': 'application/json',
            ...corsHeaders,
          },
        }
      );
    }
  },
};

/**
 * 处理 POST /api/v2/ingest/submission
 */
async function handleIngestSubmission(request, corsHeaders) {
  const url = new URL(request.url);
  const susySubmissionId = url.searchParams.get('susy_submission_id');
  const force = url.searchParams.get('force') === 'true';

  // 验证参数
  if (!susySubmissionId) {
    return new Response(
      JSON.stringify({
        error: 'Bad Request',
        message: 'susy_submission_id is required',
      }),
      {
        status: 400,
        headers: {
          'Content-Type': 'application/json',
          ...corsHeaders,
        },
      }
    );
  }

  // 生成 16 位 hash，尾数与 submission_id 一致
  const lastDigit = susySubmissionId.toString().slice(-1);
  const hash = generateMockHash16(susySubmissionId, lastDigit);

  // 模拟响应数据
  const responseData = {
    date: new Date().toISOString(),
    note: null,
    result: {
      document_hashes: [hash],
      count: 1,
    },
  };

  return new Response(JSON.stringify(responseData), {
    status: 200,
    headers: {
      'Content-Type': 'application/json',
      'Content-Length': JSON.stringify(responseData).length.toString(),
      'Server': 'uvicorn',
      ...corsHeaders,
    },
  });
}

/**
 * 处理 GET /api/v2/documents/{document_hash}/status
 */
async function handleDocumentStatus(documentHash, corsHeaders) {
  // 获取 hash 尾数
  const lastDigit = documentHash.slice(-1);
  
  let status;
  
  if (lastDigit === '0') {
    // 尾数是 0：随机返回 pending, in_progress, failed
    const statuses = ['pending', 'in_progress', 'failed'];
    status = statuses[Math.floor(Math.random() * statuses.length)];
  } else {
    // 尾数不是 0：返回 done
    status = 'done';
  }

  const responseData = {
    date: new Date().toISOString(),
    note: null,
    result: {
      status,
    },
  };

  return new Response(JSON.stringify(responseData), {
    status: 200,
    headers: {
      'Content-Type': 'application/json',
      ...corsHeaders,
    },
  });
}

/**
 * 处理 GET /api/v2/reports/{document_hash}/issues_summary
 */
async function handleIssuesSummary(documentHash, corsHeaders) {
  // 获取 hash 尾数
  const lastDigit = documentHash.slice(-1);
  
  // 尾数是 9：返回错误
  if (lastDigit === '9') {
    return new Response(
      JSON.stringify({
        error: 'Internal Server Error',
        message: 'Failed to retrieve issues summary',
        details: 'Document analysis service unavailable',
      }),
      {
        status: 500,
        headers: {
          'Content-Type': 'application/json',
          ...corsHeaders,
        },
      }
    );
  }
  
  // 尾数是 0：返回空 result
  if (lastDigit === '0') {
    const responseData = {
      date: new Date().toISOString(),
      note: null,
      result: {},
    };
    
    return new Response(JSON.stringify(responseData), {
      status: 200,
      headers: {
        'Content-Type': 'application/json',
        ...corsHeaders,
      },
    });
  }
  
  // 初始化所有字段为无问题状态
  let authors = { warning_count: 0, severe_warning_count: 0 };
  let references = { warning_count: 0, severe_warning_count: 0 };
  let aiWriting = {
    severe_warning_count: 0,
    proportions: {
      human_proportion: 1.0,
      maybe_ai_proportion: 0.0,
      likely_ai_proportion: 0.0,
    },
  };
  let peerReviewReports = [
    {
      human_proportion: 1.0,
      maybe_ai_proportion: 0.0,
      likely_ai_proportion: 0.0,
    },
  ];
  
  // 根据尾数设置问题
  switch (lastDigit) {
    case '1':
      // authors 有问题
      authors = generateWarnings();
      break;
      
    case '2':
      // references 有问题
      references = generateWarnings();
      break;
      
    case '3':
      // ai_writing 有问题，human < 0.7
      aiWriting = generateAIIssue();
      break;
      
    case '4':
      // peer_review_reports 有问题，human < 0.7
      peerReviewReports = [generatePeerReviewIssue()];
      break;
      
    case '5':
      // 随机 2 个问题
      const issues5 = getRandomIssues(2);
      if (issues5.includes('authors')) authors = generateWarnings();
      if (issues5.includes('references')) references = generateWarnings();
      if (issues5.includes('ai_writing')) aiWriting = generateAIIssue();
      if (issues5.includes('peer_review')) peerReviewReports = [generatePeerReviewIssue()];
      break;
      
    case '6':
      // 随机 3 个问题
      const issues6 = getRandomIssues(3);
      if (issues6.includes('authors')) authors = generateWarnings();
      if (issues6.includes('references')) references = generateWarnings();
      if (issues6.includes('ai_writing')) aiWriting = generateAIIssue();
      if (issues6.includes('peer_review')) peerReviewReports = [generatePeerReviewIssue()];
      break;
      
    case '7':
      // 所有问题
      authors = generateWarnings();
      references = generateWarnings();
      aiWriting = generateAIIssue();
      peerReviewReports = [generatePeerReviewIssue()];
      break;
      
    case '8':
      // 无问题（默认值已设置）
      break;
  }
  
  const responseData = {
    date: new Date().toISOString(),
    note: null,
    result: {
      issues_summary: {
        authors,
        references,
        ai_writing: aiWriting,
        peer_review_reports: peerReviewReports,
      },
      report_link: 'https://deployment.susy.mdpi.dev/',
    },
  };

  return new Response(JSON.stringify(responseData), {
    status: 200,
    headers: {
      'Content-Type': 'application/json',
      ...corsHeaders,
    },
  });
}

/**
 * 生成 16 位模拟哈希值，尾数与指定数字一致
 */
function generateMockHash16(seed, lastDigit) {
  const hash = Array.from({ length: 15 }, (_, i) => {
    const char = ((seed.charCodeAt(i % seed.length) + i) % 16).toString(16);
    return char;
  }).join('');
  return hash + lastDigit;
}

/**
 * 生成随机警告数（authors 或 references）
 */
function generateWarnings() {
  const hasWarning = Math.random() > 0.5;
  const hasSevere = Math.random() > 0.5;
  
  return {
    warning_count: hasWarning ? Math.floor(Math.random() * 5) + 1 : 0,
    severe_warning_count: hasSevere ? Math.floor(Math.random() * 3) + 1 : 0,
  };
}

/**
 * 生成 AI 写作问题（human < 0.7）
 */
function generateAIIssue() {
  const humanProportion = 0.3 + Math.random() * 0.39; // 0.3 - 0.69
  const remaining = 1.0 - humanProportion;
  const maybeAIProportion = Math.random() * remaining;
  const likelyAIProportion = remaining - maybeAIProportion;
  
  return {
    severe_warning_count: Math.floor(Math.random() * 3) + 1,
    proportions: {
      human_proportion: parseFloat(humanProportion.toFixed(2)),
      maybe_ai_proportion: parseFloat(maybeAIProportion.toFixed(2)),
      likely_ai_proportion: parseFloat(likelyAIProportion.toFixed(2)),
    },
  };
}

/**
 * 生成评审报告问题（human < 0.7）
 */
function generatePeerReviewIssue() {
  const humanProportion = 0.3 + Math.random() * 0.39; // 0.3 - 0.69
  const remaining = 1.0 - humanProportion;
  const maybeAIProportion = Math.random() * remaining;
  const likelyAIProportion = remaining - maybeAIProportion;
  
  return {
    human_proportion: parseFloat(humanProportion.toFixed(2)),
    maybe_ai_proportion: parseFloat(maybeAIProportion.toFixed(2)),
    likely_ai_proportion: parseFloat(likelyAIProportion.toFixed(2)),
  };
}

/**
 * 随机选择 N 个问题类型
 */
function getRandomIssues(count) {
  const allIssues = ['authors', 'references', 'ai_writing', 'peer_review'];
  const shuffled = allIssues.sort(() => Math.random() - 0.5);
  return shuffled.slice(0, count);
}
```

## 🌐 配置自定义域名（可选）

如果你想使用自己的域名（例如 `api.yourdomain.com`），需要先在 Cloudflare 托管你的域名。

### 步骤 1: 添加自定义域名

1. 在 Worker 详情页面，点击 **Settings** 标签
2. 找到 **Domains & Routes** 部分
3. 点击 **Add Custom Domain** 按钮
4. 输入你的子域名，例如：`api.yourdomain.com`
5. 点击 **Add Custom Domain**

### 步骤 2: 等待 DNS 配置

Cloudflare 会自动配置 DNS 记录，通常几分钟内生效。

### 步骤 3: 测试自定义域名

```bash
curl -X POST "https://api.yourdomain.com/api/v2/ingest/submission?susy_submission_id=12345"
```

## 🆘 常见问题

### Q: 部署后无法访问？
A: 检查 URL 是否正确，确保包含完整路径，例如：
```
https://your-worker.workers.dev/api/v2/documents/test123/status
```

### Q: 如何查看错误日志？
A: 
1. 在 Worker 详情页面，点击 **Logs** 标签
2. 点击 **Begin log stream** 查看实时日志

### Q: 如何删除 Worker？
A: 
1. 进入 Workers & Pages 页面
2. 点击 Worker 名称
3. 点击 **Settings** 标签
4. 滚动到底部，点击 **Delete** 按钮

### Q: 可以设置环境变量吗？
A: 可以。在 Worker 的 **Settings** > **Variables** 中添加环境变量。
